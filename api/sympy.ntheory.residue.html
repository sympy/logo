def totient_(n):
returns the number of integers less than n and relatively prime to n
def n_order(a, n):
returns the order of a modulo n Order of a modulo n is the smallest integer k such that a^k leaves a remainder of 1 with n.
def is_primitive_root(a, p):
returns True if a is a primitive root of p
def is_quad_residue(a, p):
returns True if a is a quadratic residue of p p should be a prime and a should be relatively prime to p
def legendre_symbol(a, p):
return 1 if a is a quadratic residue of p else return -1 p should be an odd prime by definition
